function pXL(file){

  return new Promise((res,rej)=>{

    const r=new FileReader();

    r.onload=e=>{

      try{

        const wb=XLSX.read(e.target.result,{type:"array"});

        const ws=wb.Sheets[wb.SheetNames[0]];

        const raw=XLSX.utils.sheet_to_json(ws,{defval:""});

        res(raw.map(r=>({

          plant:r["Plant"]||"",ageing:Number(r["Ageing(Days)"])||0,

          destination:r["Destination"]||"",customer:r["Cust.Name"]||"",

          pieces:Number(r["No of Pieces"])||0,unrestricted:parseFloat(r["Unrestricted"])||0,

}
